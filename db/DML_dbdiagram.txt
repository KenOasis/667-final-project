table users{
  id serial [pk, increment]
  username varchar(20) [unique, not null]
  password varchar(50) [not null]
  profile_url varchar(255) [not null, default: ""]
  created_at timestamp
  game_play integer [not null, default: 0]
  game_win integer [not null, default: 0]
  points integer [not null, default: 300]
}

table cards{
  id serial [pk, increment]
  type card_types [not null]
  color card_colors [not null]
  action card_actions [not null]
  face_value card_face_values [not null]
}

table games{
  id serial [pk, increment]
  created_at timestamp [not null, default: 'current']
  finished_at timestamp [not null, default: 'current']
}

table game_users{
  // unique(game_id, user_id)
  id serial [pk, increment]
  game_id integer [ref: > games.id]
  user_id integer [ref: > users.id]
  points integer [not null, default: 0]
}

table decks{
// unique(game_id, user_id)
  id serial [pk, increment]
  game_id integer [ref: > games.id]
  user_id integer [ref: > users.id]
}

table deck_cards{
// unique (deck_id, card_id)
  id serial [pk, increment]
  deck_id integer [ref: > decks.id]
  card_id integer [ref: > cards.id]
}

enum card_face_values{
  none
  zero
  one
  two
  three
  four
  five
  six
  seven
  eight
  nine
}


enum card_types{
  number [note: 'NUMBER CARD']
  action
  wild
}

enum card_actions{
  no_action
  skip
  reverse
  draw_two
  wild
  wild_draw_four
}

enum card_colors{
  red
  yellow
  green
  blue
}
